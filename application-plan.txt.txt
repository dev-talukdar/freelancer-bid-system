Use:

const hasJobFilters = profile.jobIds.length > 0;
const hasKeywordFilters = profile.keywords.length > 0;

  hasJobFilters &&
  project.jobs.some((job) =>
    profile.jobIds.includes(job.id),
  );

const keywordMatch =
  hasKeywordFilters &&
  profile.keywords.some((keyword) => {
    // normalized title/description/skill-name fallback
  });

const relevanceMatch =
  (!hasJobFilters && !hasKeywordFilters) ||
  jobMatch ||
  keywordMatch;
